```python
import hashlib, json as _json

LASTMOD_MAP_PATH = os.path.join(os.path.dirname(os.path.abspath(__file__)), "lastmod_map.json")

def load_lastmod_map():
    try:
        with open(LASTMOD_MAP_PATH, encoding="utf-8") as f:
            return _json.load(f)
    except (FileNotFoundError, ValueError):
        return {}

def save_lastmod_map(m):
    with open(LASTMOD_MAP_PATH, "w", encoding="utf-8") as f:
        _json.dump(m, f, indent=0, sort_keys=True)

def lastmod_for(m, url, content, today):
    h = hashlib.sha1(content.encode("utf-8")).hexdigest()
    prev = m.get(url)
    if prev and prev.get("hash") == h:
        return prev["date"]
    m[url] = {"hash": h, "date": today}
    return today
```

## Task 6 (BD): Fix pagination duplicates + recursive manifest pruning

```js
let slugMap = {};
export async function loadSlugMap(url = 'assets/deal-slugs.json') {
  try { const r = await fetch(url); if (r.ok) slugMap = await r.json(); }

export function _setSlugMap(m) { slugMap = m; }
```
